To get the range element working on Form2 you will need to add the following to the body of your HTML.

<input type="range" name="range" id="range" min="1" max="10" /></label><input id="rvalue" placeholder="more urgent" readonly />


<script type="text/javascript">
    var range;
    var rvalue;
    function init(){
        range = document.getElementById("range");
	rvalue = document.getElementById("rvalue");
	rvalue.placeholder = "10 is most urgent";
	range.onchange = update;
    }

    function update(){
	rvalue.value = range.value;
    }
window.onload = init;
</script>
